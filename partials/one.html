DOM Queries

Select id

var el = document.getElementById('id');

Select class

var el = document.getElementsByClassName('class');

Select single CSS

var el = document.querySelector('css selector');

Select HTML elements

var el = document.getElementsByTagName('tagName');

Select all CSS

var el = document.querySelectorAll('css selector');

Events

Event Handler

element.onevent = functionName;

Event listener

element.addEventListener('event', functionName, Boolean);

Event Handler with parameters

element.onevent = function() {
    functionName(args);
};

Event listener with parameters

element.addEventListener('event', function() {
    functionName(args);
}, Boolean);
